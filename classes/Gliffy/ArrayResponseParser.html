    def self.from_xml(element)
      single_classname = self.to_s.gsub(/s$/,'').gsub(/Gliffy::/,'')
      klass = Gliffy.const_get(single_classname)
      list = Array.new
      if (element)
        element.each_element do |element|
          list << klass.from_xml(element)
        end
      end
      list
    end
