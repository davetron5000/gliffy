    def get_users(folder_path=nil)
      url = ''
      if (folder_path)
        folder_path = Folder.encode_path_elements(folder_path)
        url += "folders/#{folder_path}/"
      end
      url += 'users'
      do_simple_rest(:get,url,"Getting users for " + (folder_path ? "folder #{folder_path}" : 'account'))
    end
